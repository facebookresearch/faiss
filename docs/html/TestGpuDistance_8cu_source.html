20 void testTransposition(bool colMajorVecs,
21  bool colMajorQueries,
22  faiss::MetricType metric) {
23  int device = faiss::gpu::randVal(0, faiss::gpu::getNumDevices() - 1);
24 
25  faiss::gpu::StandardGpuResources res;
26  res.noTempMemory();
27 
28  int dim = faiss::gpu::randVal(20, 150);
29  int numVecs = faiss::gpu::randVal(10, 30000);
30  int numQuery = faiss::gpu::randVal(1, 1024);
31  int k = faiss::gpu::randVal(20, 70);
32 
33  // Input data for CPU
34  std::vector<float> vecs = faiss::gpu::randVecs(numVecs, dim);
35  std::vector<float> queries = faiss::gpu::randVecs(numQuery, dim);
36 
37  // The CPU index is our reference for the results
38  faiss::IndexFlatL2 cpuIndexL2(dim);
39  cpuIndexL2.add(numVecs, vecs.data());
40 
41  std::vector<float> cpuDistanceL2(numQuery * k, 0);
42  std::vector<faiss::Index::idx_t> cpuIndicesL2(numQuery * k, -1);
43 
44  cpuIndexL2.search(numQuery, queries.data(), k,
45  cpuDistanceL2.data(), cpuIndicesL2.data());
46 
47  faiss::IndexFlatIP cpuIndexIP(dim);
48  cpuIndexIP.add(numVecs, vecs.data());
49 
50  std::vector<float> cpuDistanceIP(numQuery * k, 0);
51  std::vector<faiss::Index::idx_t> cpuIndicesIP(numQuery * k, -1);
52 
53  cpuIndexIP.search(numQuery, queries.data(), k,
54  cpuDistanceIP.data(), cpuIndicesIP.data());
55 
56  // The transpose and distance code assumes the desired device is already set
57  faiss::gpu::DeviceScope scope(device);
58  auto stream = res.getDefaultStream(device);
59 
60  // Copy input data to GPU, and pre-transpose both vectors and queries for
61  // passing
62  auto gpuVecs = faiss::gpu::toDevice<float, 2>(
63  nullptr, device, vecs.data(), stream, {numVecs, dim});
64  auto gpuQueries = faiss::gpu::toDevice<float, 2>(
65  nullptr, device, queries.data(), stream, {numQuery, dim});
66 
67  faiss::gpu::DeviceTensor<float, 2, true> vecsT({dim, numVecs});
68  faiss::gpu::runTransposeAny(gpuVecs, 0, 1, vecsT, stream);
69 
70  faiss::gpu::DeviceTensor<float, 2, true> queriesT({dim, numQuery});
71  faiss::gpu::runTransposeAny(gpuQueries, 0, 1, queriesT, stream);
72 
73  std::vector<float> gpuDistance(numQuery * k, 0);
74  std::vector<faiss::Index::idx_t> gpuIndices(numQuery * k, -1);
75 
76  faiss::gpu::bruteForceKnn(
77  &res,
78  metric,
79  colMajorVecs ? vecsT.data() : gpuVecs.data(),
80  !colMajorVecs,
81  numVecs,
82  colMajorQueries ? queriesT.data() : gpuQueries.data(),
83  !colMajorQueries,
84  numQuery,
85  dim,
86  k,
87  gpuDistance.data(),
88  gpuIndices.data());
89 
90  std::stringstream str;
91  str << "metric " << metric
92  << " colMajorVecs " << colMajorVecs
93  << " colMajorQueries " << colMajorQueries;
94 
95  faiss::gpu::compareLists(metric == faiss::MetricType::METRIC_L2 ?
96  cpuDistanceL2.data() : cpuDistanceIP.data(),
97  metric == faiss::MetricType::METRIC_L2 ?
98  cpuIndicesL2.data() : cpuIndicesIP.data(),
99  gpuDistance.data(),
100  gpuIndices.data(),
101  numQuery, k,
102  str.str(),
103  false, false, true,
104  6e-3f, 0.1f, 0.015f);
105 }
