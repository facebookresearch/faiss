104  /**************************************
105  * Scoped inverted lists (for automatic deallocation)
106  *
107  * instead of writing:
108  *
109  * uint8_t * codes = invlists->get_codes (10);
110  * ... use codes
111  * invlists->release_codes(codes)
112  *
113  * write:
114  *
115  * ScopedCodes codes (invlists, 10);
